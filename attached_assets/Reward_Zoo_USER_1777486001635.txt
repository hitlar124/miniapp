        async function handleAuthStateChange(user) {
            if (user) {
                loader.style.display = 'flex';
                currentUser = user;
                try {
                    await fetchAppConfig();
                    await fetchUserData();
                    if (userData && userData.isBlocked) { showAlert("Your account has been blocked by the admin."); await signOut(auth); return; }
                    
                    // --- AUTO REFERRAL CHECK AFTER SUCCESSFUL LOGIN ---
                    if (window.Telegram && window.Telegram.WebApp && window.Telegram.WebApp.initDataUnsafe) {
                        const startParam = window.Telegram.WebApp.initDataUnsafe.start_param;
                        if (startParam && !userData.referredBy && startParam !== userData.referralCode) {
                            await autoApplyReferral(startParam);
                        }
                    }

                    await checkNewNotifications();
                    updateUI();
                    startCooldownTimers();
                    showMainPage('app-container');
                    navigateTo('home-page');
                } catch (error) { console.error("Error:", error); showAlert("Failed to load data."); } 
                finally { loader.style.display = 'none'; }
            } else {
                currentUser = null; userData = null;
                if(timerInterval) clearInterval(timerInterval);
                showMainPage('auth-section'); loader.style.display = 'none';
            }
        }

        async function autoApplyReferral(code) {
            const bonus = appConfig.referralBonus || 100;
            try {
                const snap = await getDocs(query(collection(db, "users"), where("referralCode", "==", code)));
                if (!snap.empty) {
                    const refDoc = snap.docs[0];
                    await updateDoc(doc(db, "users", refDoc.id), { balance: increment(bonus) });
                    await updateDoc(doc(db, "users", currentUser.uid), { balance: increment(bonus), referredBy: code });
                    userData.balance += bonus; 
                    userData.referredBy = code;
                    updateUI();
                    showAlert(`🎉 You received ${bonus} coins as sign-up bonus from the invite link!`);
                }
            } catch(e) { console.error("Auto Refer Error", e); }
        }

                    dailyAdLimit: data.dailyAdLimit || 10,

        async function fetchUserData() {
            if (!currentUser) return;
            const userRef = doc(db, "users", currentUser.uid);
            const userSnap = await getDoc(userRef);
            const todayStr = new Date().toISOString().split('T')[0];

            if (userSnap.exists()) {
                userData = userSnap.data();
                let updatesNeeded = false; let updateData = {};
                
                if (userData.lastAdWatchDate !== todayStr) { updateData.dailyAdCount = 0; updateData.lastAdWatchDate = todayStr; userData.dailyAdCount = 0; userData.lastAdWatchDate = todayStr; updatesNeeded = true; }
                if (userData.lastMathDate !== todayStr) { updateData.dailyMathCount = 0; updateData.lastMathDate = todayStr; userData.dailyMathCount = 0; userData.lastMathDate = todayStr; updatesNeeded = true; }
                
                if (typeof userData.lastMathTime === 'undefined') { updateData.lastMathTime = 0; userData.lastMathTime = 0; updatesNeeded = true; }
                if (typeof userData.lastAdTime === 'undefined') { updateData.lastAdTime = 0; userData.lastAdTime = 0; updatesNeeded = true; }
                if (typeof userData.lastCheckInDate === 'undefined') { updateData.lastCheckInDate = null; userData.lastCheckInDate = null; updatesNeeded = true; }
                
                if (updatesNeeded) await updateDoc(userRef, updateData);
            } else {
                const ownRefCode = Math.random().toString(36).substring(2, 8).toUpperCase();
                const defaultUserData = {
                    uid: currentUser.uid, name: currentUser.displayName || currentUser.email.split('@')[0], email: currentUser.email,
                    balance: 50, referralCode: ownRefCode, referredBy: null, lastNotificationCheck: serverTimestamp(), createdAt: serverTimestamp(),
                    dailyAdCount: 0, lastAdWatchDate: todayStr, lastAdTime: 0,
                    dailyMathCount: 0, lastMathDate: todayStr, lastMathTime: 0,
                    lastCheckInDate: null, isBlocked: false
                };
                await setDoc(userRef, defaultUserData);
                const freshSnap = await getDoc(userRef); userData = freshSnap.data();
            }
        }

        function startCooldownTimers() {
            if(timerInterval) clearInterval(timerInterval);
            timerInterval = setInterval(updateTasksUI, 1000); 
        }

        function formatTime(ms) {
            const totalSec = Math.ceil(ms / 1000);
            const m = Math.floor(totalSec / 60).toString().padStart(2, '0');
            const s = (totalSec % 60).toString().padStart(2, '0');
            return `${m}:${s}`;
        }

        function updateTasksUI() {
            if(!userData || !appConfig) return;
            const today = new Date().toISOString().split('T')[0];
            const now = Date.now();
            
            const btnCheckin = document.getElementById('btn-checkin');
            if (userData.lastCheckInDate === today) {
                btnCheckin.textContent = 'Claimed';
                btnCheckin.className = "px-5 py-2 rounded-lg text-sm font-bold btn-disabled";
            } else {
                btnCheckin.textContent = 'Claim';
                btnCheckin.className = "px-5 py-2 rounded-lg text-sm font-bold btn-neon";
            }
            
            const mathLimit = appConfig.dailyMathLimit || 10;
            const mathLeft = Math.max(0, mathLimit - (userData.dailyMathCount || 0));
            document.getElementById('math-left-count').textContent = mathLeft;
            
            const mathCooldownMs = (appConfig.mathCooldown || 60) * 1000;
            const mathTimeLeft = mathCooldownMs - (now - (userData.lastMathTime || 0));
            const btnMath = document.getElementById('btn-math');
            
            if (mathLeft <= 0) {
                btnMath.textContent = 'Done';
                btnMath.className = "px-5 py-2 rounded-lg text-sm font-bold btn-disabled";
            } else if (mathTimeLeft > 0) {
                btnMath.textContent = formatTime(mathTimeLeft);
                btnMath.className = "px-5 py-2 rounded-lg text-sm font-bold btn-disabled tracking-widest";
            } else {
                btnMath.textContent = 'Play';
                btnMath.className = "px-5 py-2 rounded-lg text-sm font-bold btn-neon";
            }

            const adLimit = appConfig.dailyAdLimit || 10;
            const adsLeft = Math.max(0, adLimit - (userData.dailyAdCount || 0));
            document.getElementById('ads-left-count').textContent = adsLeft;

            const adCooldownMs = (appConfig.adCooldown || 60) * 1000;
            const adTimeLeft = adCooldownMs - (now - (userData.lastAdTime || 0));
            const btnAd = document.getElementById('btn-ad');
            
            if (adsLeft <= 0) {
                btnAd.textContent = 'Done';
                btnAd.className = "px-5 py-2 rounded-lg text-sm font-bold btn-disabled";
            } else if (adTimeLeft > 0) {
                btnAd.textContent = formatTime(adTimeLeft);
                btnAd.className = "px-5 py-2 rounded-lg text-sm font-bold btn-disabled tracking-widest";
            } else {
                btnAd.textContent = 'Watch';
                btnAd.className = "px-5 py-2 rounded-lg text-sm font-bold btn-neon";
            }
        }

        window.openMathQuiz = function() {
            const now = Date.now();
            const mathCooldownMs = (appConfig.mathCooldown || 60) * 1000;
            if (userData.dailyMathCount >= (appConfig.dailyMathLimit || 10)) return showAlert("Daily limit reached! Come back tomorrow.");
            if (now - (userData.lastMathTime || 0) < mathCooldownMs) return;

            const num1 = Math.floor(Math.random() * 20) + 1; const num2 = Math.floor(Math.random() * 20) + 1; const isPlus = Math.random() > 0.5;
            if (isPlus) { currentMathAnswer = num1 + num2; document.getElementById('math-question').textContent = `${num1} + ${num2} = ?`; } 
            else { const mx = Math.max(num1, num2); const mn = Math.min(num1, num2); currentMathAnswer = mx - mn; document.getElementById('math-question').textContent = `${mx} - ${mn} = ?`; }
            
            document.getElementById('math-answer').value = '';
            document.getElementById('math-quiz-modal').classList.remove('hidden');
        }

        window.closeMathQuiz = function() { document.getElementById('math-quiz-modal').classList.add('hidden'); }

                loader.style.display = 'flex';

        }

                    const refDoc = snap.docs[0];
                    await updateDoc(doc(db, "users", refDoc.id), { balance: increment(bonus) });
                    await updateDoc(doc(db, "users", currentUser.uid), { balance: increment(bonus), referredBy: code });
