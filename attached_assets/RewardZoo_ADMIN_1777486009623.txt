                            <button class="px-3 py-1.5 rounded-lg text-sm font-semibold bg-violet-500/10 text-violet-400 border border-violet-500/20 hover:bg-violet-500/20 transition" onclick="window.openEditBalance('${doc.id}', '${nameStr.replace(/'/g, "\\'")}', ${user.balance || 0})">Edit Coins</button>

                });

        }

        function loadWithdrawals() {
            const wTableBody = document.getElementById('withdrawals-table-body');
            onSnapshot(query(collection(db, "withdrawals"), where("status", "==", "pending")), (snapshot) => {
                wTableBody.innerHTML = '';
                snapshot.forEach(docSnap => {
                    const request = docSnap.data();
                    const requestedDate = request.requestedAt?.toDate().toLocaleString() || 'N/A';
                    const row = `<tr>
                        <td class="font-medium">${request.userName || 'N/A'}<br><span class="text-xs text-gray-500">${request.userEmail}</span></td>
                        <td class="font-bold text-fuchsia-400">${request.amount}</td>
                        <td class="uppercase text-sm tracking-wide text-gray-300">${request.method}</td>
                        <td class="text-white">${request.paymentDetail || 'N/A'}</td>
                        <td class="text-xs text-gray-400">${requestedDate}</td>
                        <td class="flex items-center gap-2">
                            <button class="px-3 py-1.5 rounded-lg text-sm font-semibold btn-success" onclick="window.handleWithdrawal('${docSnap.id}', 'approved')">Approve</button>
                            <button class="px-3 py-1.5 rounded-lg text-sm font-semibold btn-danger" onclick="window.handleWithdrawal('${docSnap.id}', 'rejected', '${request.userId}', ${request.amount})">Reject</button>
                        </td>
                    </tr>`;
                    wTableBody.innerHTML += row;
                });
            });
        }

        async function loadSettings() {
            const configRef = doc(db, "config", "main");
            const docSnap = await getDoc(configRef);
            if (docSnap.exists()) {
                const config = docSnap.data();
                
                // Limits & Cooldowns
                document.getElementById('daily-ad-limit').value = config.dailyAdLimit || 10;
                document.getElementById('ad-cooldown').value = config.adCooldown || 60; // default 60 sec
                document.getElementById('daily-math-limit').value = config.dailyMathLimit || 10;
                document.getElementById('math-cooldown').value = config.mathCooldown || 60; // default 60 sec
                
                // Rewards
                document.getElementById('checkin-reward').value = config.checkinReward || 50;
                document.getElementById('referral-bonus').value = config.referralBonus || 100;
                document.getElementById('ad-reward').value = config.adReward || 50;
                document.getElementById('math-reward').value = config.mathReward || 10;

        // Action Handlers
        window.openEditBalance = (userId, userName, currentBalance) => {
            currentEditUserId = userId;
            document.getElementById('edit-balance-user-name').textContent = `Editing limits for: ${userName}`;
            document.getElementById('new-balance-input').value = currentBalance;
            document.getElementById('balance-modal').classList.remove('hidden');
        };

        window.closeBalanceModal = () => {
            document.getElementById('balance-modal').classList.add('hidden');
            currentEditUserId = null;
        };

        window.toggleBlockUser = (userId, isBlocked) => {
            const action = isBlocked ? 'unblock' : 'block';
            showConfirm(`Are you sure you want to ${action} this user?`, async () => {
                const userRef = doc(db, "users", userId);
                try {
                    await updateDoc(userRef, { isBlocked: !isBlocked });
                    showAlert(`User successfully ${action}ed.`);
                } catch (error) { showAlert(`Error: ${error.message}`); }
            });
        };

        window.handleWithdrawal = (requestId, newStatus, userId, amount) => {
            showConfirm(`Are you sure you want to ${newStatus} this request?`, async () => {
                const requestRef = doc(db, "withdrawals", requestId);
                try {
                    if (newStatus === 'rejected') {
                        const userRef = doc(db, "users", userId);
                        await updateDoc(userRef, { balance: increment(amount) });
                    }
                    await updateDoc(requestRef, { status: newStatus });
                    showAlert(`Request has been ${newStatus}.`);
                } catch (error) { showAlert(`Error: ${error.message}`); }
            });
        };

        async function sendNotification() {
            const title = document.getElementById('notification-title').value.trim();
            const message = document.getElementById('notification-message').value.trim();
            if (!title || !message) return showAlert("Both Title and Message are required.");
            try {
                await addDoc(collection(db, "notifications"), { title, message, createdAt: serverTimestamp() });
                showAlert("Notification broadcasted successfully to all users!");
                document.getElementById('notification-title').value = '';
                document.getElementById('notification-message').value = '';
            } catch (error) { showAlert(`Error: ${error.message}`); }
        }

        async function saveSettings() {
            const settingsData = {
                // Limits & Cooldowns
                dailyAdLimit: parseInt(document.getElementById('daily-ad-limit').value) || 10,
                adCooldown: parseInt(document.getElementById('ad-cooldown').value) || 60,
                dailyMathLimit: parseInt(document.getElementById('daily-math-limit').value) || 10,
                mathCooldown: parseInt(document.getElementById('math-cooldown').value) || 60,
                
                // Rewards
                checkinReward: parseInt(document.getElementById('checkin-reward').value) || 50,
                referralBonus: parseInt(document.getElementById('referral-bonus').value) || 100,
                adReward: parseInt(document.getElementById('ad-reward').value) || 50,
                mathReward: parseInt(document.getElementById('math-reward').value) || 10,

                // Finance
                minWithdrawal: parseInt(document.getElementById('min-withdrawal').value) || 5000,
                coinValueCoins: parseInt(document.getElementById('coin-value-coins').value) || 1000,
                coinValueInr: parseInt(document.getElementById('coin-value-inr').value) || 10,
                paymentMethods: document.getElementById('payment-methods').value.split(',').map(s => s.trim()).filter(Boolean)
            };

            try {
                await setDoc(doc(db, "config", "main"), settingsData, { merge: true });
                showAlert("All configurations have been successfully saved!");
            } catch (error) { showAlert(`Error saving settings: ${error.message}`); }
        }
        
        function showAlert(message) {
            document.getElementById('alert-message').textContent = message;
            document.getElementById('custom-alert').classList.remove('hidden');
            document.getElementById('alert-ok-btn').onclick = () => { document.getElementById('custom-alert').classList.add('hidden'); };
        }

        function showConfirm(message, onConfirm) {
            const confirmModal = document.getElementById('custom-confirm');
            const confirmMessage = document.getElementById('confirm-message');
            const confirmOkBtn = document.getElementById('confirm-ok-btn');
            const confirmCancelBtn = document.getElementById('confirm-cancel-btn');

            confirmMessage.textContent = message;
            confirmModal.classList.remove('hidden');

            const okListener = () => {
                onConfirm();
                confirmModal.classList.add('hidden');
                cleanup();
            };

            const cancelListener = () => {
                confirmModal.classList.add('hidden');
                cleanup();
            };
            
            const cleanup = () => {
                confirmOkBtn.removeEventListener('click', okListener);
                confirmCancelBtn.removeEventListener('click', cancelListener);
            }

            confirmOkBtn.addEventListener('click', okListener);
            confirmCancelBtn.addEventListener('click', cancelListener);
        }
